119 -(void)toggleWiggleAnimation:(BOOL)animate {
120  if (animate) {
121  CATransform3D transform;
122 
123  if (arc4random() % 2 == 1)
124  transform = CATransform3DMakeRotation(-0.08, 0, 0, 1.0);
125  else
126  transform = CATransform3DMakeRotation(0.08, 0, 0, 1.0);
127  CABasicAnimation *animation = [CABasicAnimation animationWithKeyPath:@"transform"];
128  animation.toValue = [NSValue valueWithCATransform3D:transform];
129  animation.duration = 0.2;
130  animation.repeatCount = NSIntegerMax;
131  animation.autoreverses = YES;
132  animation.delegate = self;
133  [[self layer] addAnimation:animation forKey:@"wiggleAnimation"];
134  } else {
