147 -(NSInteger)numberOfItemsInSpringboard:(PTSSpringBoard *)springboard {
148  return [[self items] count];

151 -(PTSSpringBoardItem*)springboard:(PTSSpringBoard *)springboard itemForIndex:(NSInteger)index {
152  PTSSpringBoardItem * item = [[PTSSpringBoardItem alloc] init];
153 
154  [item setLabel:[[[self items] objectAtIndex:index] objectForKey:@"label"]];
155  [item setIcon:[[[self items] objectAtIndex:index] objectForKey:@"icon"]];
156  [item setItemIdentifier:[[[self items] objectAtIndex:index] objectForKey:@"identifier"]];
157  [item setBadgeValue:[NSString stringWithFormat:@"%d", (index + 1)]];
158 
159  return item;
160 }

193  UIAlertView * alert = [[UIAlertView alloc] initWithTitle:@"Entered editing mode..." message:[NSString stringWithFormat:@"You taped the item at position %d. It uses the item-identifier '%@'.", (index + 1), [item itemIdentifier]] delegate:self cancelButtonTitle:@"OK" otherButtonTitles:nil];
