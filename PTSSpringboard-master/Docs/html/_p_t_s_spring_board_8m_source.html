75 -(void)layoutSubviews {
76  /*Layouts the item container...*/
77  itemsContainer.frame = CGRectMake(0.0f, 0.0f, self.frame.size.width, self.frame.size.height);
78 
79  /*...the page-control.*/
80  pageControl.frame = CGRectMake(0.0f, self.frame.size.height-20.0f, self.frame.size.width, 20.0f);
81 
82  /*Now layouts the content-size of the item-container...*/
83  NSInteger numberOfItems = [[self delegate] numberOfItemsInSpringboard:self];
84 
85  CGSize sizePerItem = CGSizeMake(self.itemSize, self.itemSize);
86 
87  NSInteger numberOfRows = floor(self.frame.size.width / sizePerItem.width);
88  NSInteger numberOfColumns = floor(self.frame.size.height / sizePerItem.height);
89  NSInteger numberOfItemsPerPage = numberOfRows * numberOfColumns;
90  NSInteger numberOfPages = ceil((float)numberOfItems/numberOfItemsPerPage);
91 
92 
93  if (pageControl.numberOfPages > numberOfPages) {
94  pageControl.currentPage = pageControl.currentPage - 1;
95  }
96 
97  pageControl.numberOfPages = numberOfPages;
98 
99  [itemsContainer setContentSize:CGSizeMake(numberOfPages*itemsContainer.frame.size.width, itemsContainer.frame.size.height)];
100 
101  CGFloat rowGap = ([[self superview] bounds].size.width - numberOfRows * sizePerItem.width)/numberOfRows;
102  CGFloat columnGap = ([[self superview] bounds].size.height - numberOfColumns * sizePerItem.height)/numberOfColumns;
103 
104  /*...and the positions of the individual items.*/
105  for (PTSSpringBoardItem * item in itemsInOrder) {
106  NSInteger index = [itemsInOrder indexOfObject:item];
107 
108  if (index != NSNotFound) {
109  NSInteger pageNumber = floor((float)(index/numberOfItemsPerPage));
110 
111  NSInteger collumnIdx = floor((float) (index % numberOfItemsPerPage) / numberOfRows);
112  NSInteger rowIdx = (index % numberOfRows);
113 
114  /*If animation is in editing mode (except for the first layouting, when updating the springboard), repositioning of all items is animated. Otherwise it will be unanimated.*/
115  if ([self isEditing] && ![self isUpdating]) {
116  [UIView animateWithDuration:0.25f animations:^(void){
117  [item setFrame:CGRectMake((rowIdx * (self.itemSize + rowGap)) + (pageNumber*itemsContainer.frame.size.width), (collumnIdx * (self.itemSize + columnGap)), sizePerItem.width, sizePerItem.height)];
118  }];
119  } else {
120  [item setFrame:CGRectMake((rowIdx * (self.itemSize + rowGap)) + (pageNumber*itemsContainer.frame.size.width), (collumnIdx * (self.itemSize + columnGap)), sizePerItem.width, sizePerItem.height)];
121  }
122  } else {
123  NSLog(@"Error!");
124  }
125  }
126 
127  /*Sets updating to NO if it is set to YES.*/
128  if ([self isUpdating]) {
129  updating = NO;
130  }
131 
132  [super layoutSubviews];
133 }

143 -(BOOL)item:(PTSSpringBoardItem*)item collidesWithOtherItem:(PTSSpringBoardItem **)collisionItem {
144  for (PTSSpringBoardItem * spItem in itemsInOrder) {
145  if (spItem != item && [item isMovable] && [spItem isMovable]) {
146  CGPoint p1 = item.center;
147  CGPoint p2 = spItem.center;
148 
149  if (sqrt((p1.x - p2.x) * (p1.x - p2.x) + (p1.y - p2.y) * (p1.y - p2.y)) <= 20.0f) {
150  *collisionItem = spItem;
151  return YES;
152  }
153  }
154  }
155 
156  return NO;
157 }

185  PTSSpringBoardItem * item = [[self dataSource] springboard:self itemForIndex:index];
186 
187  if ([itemsInOrder count] >= (index + 1) && item) {
188  PTSSpringBoardItem * oldItem = [itemsInOrder objectAtIndex:index];

264 -(void)springboardIsDraging:(PTSSpringBoardItem *)item {
265  /*A springboard item reports that a touchEvent is moving, which will cause the corresponding item to move. This method takes care of two things:
266 
267  a) It checks for collisions with other items in the view. If one is detected, and the delegate responds to -(void)springboard:movedItemAtIndex:toIndex:, it will cause the two items to be exchanged internally.
268  b) It checks, wheather the item is move across the borders to another page.
269 
270  The internal exchange of items upon collision is only temporary for display reasons. The data source will be informed of the exchange with the -(void)springboard: movedItem: atIndex: toIndex: method. It has to take care of the reordering, otherwise the items position will be reset, when dragging ends.
271  */
272  PTSSpringBoardItem * collisionItem;
273 
274  /*Checks for collision.*/
275  if ([self item:item collidesWithOtherItem:&collisionItem] && [[self delegate] respondsToSelector:@selector(springboard:movedItem:atIndex:toIndex:)]) {
276 
277  oldIndex = [itemsInOrder indexOfObject:item];
278  newIndex = [itemsInOrder indexOfObject:collisionItem];
279 
280  if (oldIndex != NSNotFound && newIndex != NSNotFound) {
281  [itemsInOrder exchangeObjectAtIndex:newIndex withObjectAtIndex:oldIndex];
282  [[self delegate] springboard:self movedItem:item atIndex:oldIndex toIndex:newIndex];
283 
284  [self setNeedsLayout];
285  }
286  }
287 
288  /*Checks for moving an item across a border.*/
289  CGRect contentFrame = CGRectMake(0.0f, 0.0f, itemsContainer.contentSize.width, itemsContainer.contentSize.height);
290  if (!CGRectContainsRect(itemsContainer.frame, item.frame) && CGRectContainsRect(contentFrame, item.frame)) {
291 
292  CGRect visibleFrame = CGRectMake(itemsContainer.contentOffset.x, itemsContainer.contentOffset.y, itemsContainer.contentSize.width, itemsContainer.contentSize.height);
293 
294  CGRect nextPageRect;
295 
296  if (item.frame.origin.x > visibleFrame.origin.x) {
297  nextPageRect = CGRectMake(((pageControl.currentPage + 1) * itemsContainer.bounds.size.width), itemsContainer.bounds.origin.y, itemsContainer.bounds.size.width, itemsContainer.bounds.size.height);
298  } else {
299  nextPageRect = CGRectMake(((pageControl.currentPage - 1) * itemsContainer.bounds.size.width), itemsContainer.bounds.origin.y, itemsContainer.bounds.size.width, itemsContainer.bounds.size.height);
300  }
301 
302  [itemsContainer scrollRectToVisible:nextPageRect animated:YES];
303  }
